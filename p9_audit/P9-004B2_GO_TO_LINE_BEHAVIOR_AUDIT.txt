stilus_commands_find_by_id(const char *id)
{
    gsize i;

    if (id == NULL)
        return NULL;

    for (i = 0; i < G_N_ELEMENTS(stilus_commands); i++)
    {
        if (g_strcmp0(stilus_commands[i].id, id) == 0)
            return &stilus_commands[i];

    return NULL;
}
