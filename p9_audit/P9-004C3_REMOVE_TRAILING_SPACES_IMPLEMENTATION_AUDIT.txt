+static char *
+stilus_remove_trailing_spaces_from_text(const char *text)
+{
+    GString *out = g_string_new(NULL);
+    gsize i = 0;
+
+    while (text[i] != '\0')
+    {
+        gsize line_start = i;
+        gsize line_end = i;
+
+        while (text[line_end] != '\0' && text[line_end] != '\n' && text[line_end] != '\r')
+            line_end++;
+
+        gsize trimmed_end = line_end;
+        while (trimmed_end > line_start &&
+               (text[trimmed_end - 1] == ' ' || text[trimmed_end - 1] == '\t'))
+            trimmed_end--;
+
+        g_string_append_len(out, text + line_start, trimmed_end - line_start);
+
+        if (text[line_end] == '\r' && text[line_end + 1] == '\n')
+        {
+            g_string_append(out, "\r\n");
+            i = line_end + 2;
+        }
+        else if (text[line_end] == '\n' || text[line_end] == '\r')
+        {
+            g_string_append_c(out, text[line_end]);
+            i = line_end + 1;
+        }
+        else
+        {
+            i = line_end;
+        }
+    }
+
+    return g_string_free(out, FALSE);
+}
